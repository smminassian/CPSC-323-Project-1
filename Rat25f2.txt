begin
    read n;
    read m;
    sum = 0;
    i = 0;

    while (i < n)
    {
        j = 0;
        while (j < m)
        {
            sum = sum + (i * j);
            j = j + 1;
        }
        i = i + 1;
    }

    avg = sum / (n * m);

    repeat
        if (avg > 100) then
            avg = avg - 10;
        else
            avg = avg + 5;
        end
        n = n - 1;
    until (n == 0);

    write sum;
    write avg;
end
